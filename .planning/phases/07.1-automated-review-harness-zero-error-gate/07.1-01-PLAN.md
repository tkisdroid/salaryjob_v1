    ```typescript
    // Source of truth: D-17 (aggregate ALL gates before reporting), SC #1 (exit 0 smoke path)
    // This runner in Plan 01 exercises the SINGLE-ROUTE smoke path only.
    // Plan 02 Task 1 (run-full-sweep.ts) reuses the same gate functions for 108 scenarios.

    import { spawnSync } from 'node:child_process';
    import { mkdirSync, writeFileSync } from 'node:fs';

    type GateResult = { id: string; name: string; passed: boolean; observed: string; message?: string };

    async function runSmokeGates(route: string): Promise<GateResult[]> {
      const results: GateResult[] = [];

      // G6 — tsc --noEmit
      const tsc = spawnSync('npx', ['tsc', '--noEmit'], { encoding: 'utf8', shell: process.platform === 'win32' });
      results.push({ id: 'G6', name: 'TypeScript', passed: tsc.status === 0, observed: tsc.status === 0 ? '0 errors' : 'errors present', message: tsc.stderr });

      // G7 — eslint (warnings allowed)
      const eslint = spawnSync('npx', ['eslint', '.', '--max-warnings=-1'], { encoding: 'utf8', shell: process.platform === 'win32' });
      results.push({ id: 'G7', name: 'ESLint', passed: eslint.status === 0, observed: eslint.status === 0 ? '0 errors' : 'errors present', message: eslint.stderr });

      // G16 — vitest (existing suite regression baseline per SC #9)
      const vitest = spawnSync('npx', ['vitest', 'run'], { encoding: 'utf8', shell: process.platform === 'win32' });
      results.push({ id: 'G16', name: 'vitest', passed: vitest.status === 0, observed: vitest.status === 0 ? 'all pass' : 'failures present', message: vitest.stderr });

      // G1..G5 + G8 + G13 + G14 — Playwright smoke spec
      //   tests/review/smoke/root-smoke.spec.ts invokes the helpers from tests/review/helpers/
      //   and emits tests/review/_results.json. Parse it for pass/fail.
      const pw = spawnSync('npx', ['playwright', 'test', '--project=review-desktop', `--grep`, `@smoke`], {
        encoding: 'utf8', shell: process.platform === 'win32', env: { ...process.env, REVIEW_RUN: '1' },
      });
      results.push({ id: 'G1-G5,G8,G13,G14', name: 'Playwright smoke', passed: pw.status === 0, observed: pw.status === 0 ? 'pass' : 'fail', message: pw.stderr });

      // G9..G12 — Lighthouse CI
      const lhci = spawnSync('npx', ['lhci', 'autorun', `--collect.url=http://localhost:3000${route === '/' ? '' : route}`], {
        encoding: 'utf8', shell: process.platform === 'win32',
      });
      results.push({ id: 'G9-G12', name: 'Lighthouse CI', passed: lhci.status === 0, observed: lhci.status === 0 ? 'within budget' : 'budget exceeded', message: lhci.stderr });

      return results;
    }

    async function main() {
      const routeArg = process.argv.find((a) => a.startsWith('--route='));
      const route = routeArg ? routeArg.slice('--route='.length) : '/';

      mkdirSync('.review', { recursive: true });
      const results = await runSmokeGates(route);
      writeFileSync('.review/smoke-results.json', JSON.stringify(results, null, 2));

      const allPassed = results.every((r) => r.passed);
      console.log(`[review:smoke] route=${route} gates=${results.length} passed=${results.filter((r) => r.passed).length}`);
      for (const r of results) {
        console.log(`  ${r.id} ${r.name}: ${r.passed ? 'PASS' : 'FAIL'} (${r.observed})`);
      }
      process.exit(allPassed ? 0 : 1);
    }

    main().catch((err) => { console.error(err); process.exit(1); });
    ```
