    import { spawnSync } from 'node:child_process';
    import { readFileSync, writeFileSync, mkdirSync, existsSync } from 'node:fs';
    import { ROUTES } from '../../tests/review/routes/manifest';

    export type GateFailure = {
      gate: 'G1'|'G2'|'G3'|'G4'|'G5'|'G6'|'G7'|'G8'|'G9'|'G10'|'G11'|'G12'|'G13'|'G14'|'G15'|'G16';
      route?: string; viewport?: 'desktop'|'mobile-375'; persona?: 'worker'|'biz'|'admin'|'anon';
      message: string; file_suggestion?: string; stack?: string;
    };

    export type AggregateReport = {
      iteration: number;
      started_at: string;
      ended_at: string;
      failures: GateFailure[];
      flows: { id: string; name: string; passed: boolean; elapsedMs: number }[];
      gates_summary: Record<string, { observed: number | string; threshold: string; passed: boolean }>;
    };

    export async function runFullSweep(iter: number): Promise<AggregateReport> {
      const started = new Date().toISOString();
      const failures: GateFailure[] = [];
      const flows: AggregateReport['flows'] = [];
      const gates_summary: AggregateReport['gates_summary'] = {};

      // Stage 1 — static gates (G6, G7, G16)
      const tsc = spawnSync('npx', ['tsc', '--noEmit'], { encoding: 'utf8', shell: process.platform === 'win32' });
      gates_summary['G6'] = { observed: tsc.status === 0 ? 0 : 'errors', threshold: '0', passed: tsc.status === 0 };
      if (tsc.status !== 0) failures.push({ gate: 'G6', message: tsc.stdout });

      const eslint = spawnSync('npx', ['eslint', '.', '--max-warnings=-1', '--format=json'], { encoding: 'utf8', shell: process.platform === 'win32' });
      gates_summary['G7'] = { observed: eslint.status === 0 ? 0 : 'errors', threshold: '0 errors (warnings allowed)', passed: eslint.status === 0 };
      if (eslint.status !== 0) failures.push({ gate: 'G7', message: eslint.stdout });

      const vitest = spawnSync('npx', ['vitest', 'run'], { encoding: 'utf8', shell: process.platform === 'win32' });
      gates_summary['G16'] = { observed: vitest.status === 0 ? 'all pass' : 'failures', threshold: 'all pass', passed: vitest.status === 0 };
      if (vitest.status !== 0) failures.push({ gate: 'G16', message: vitest.stdout });

      // Stage 2 — 108 route-scenarios (G1..G5, G8, G13, G14) via tests/review/routes/run-matrix.ts
      const pw = spawnSync('npx',
        ['playwright', 'test', '--project=review-desktop', '--project=mobile-375'],
        { encoding: 'utf8', shell: process.platform === 'win32', env: { ...process.env, REVIEW_RUN: '1' } });
      const pwResults = existsSync('tests/review/_results.json')
        ? JSON.parse(readFileSync('tests/review/_results.json', 'utf8'))
        : null;
      // ... parse pwResults and populate failures with per-scenario G1..G5/G8/G13/G14 entries
      // ... populate gates_summary for G1..G5, G8, G13, G14 with pass/total counts

      // Stage 3 — 7 E2E flows (G15) — specs from Task 2 (tests/review/flows/*.spec.ts)
      const flowRun = spawnSync('npx',
        ['playwright', 'test', '--grep', '@flow-', '--project=review-desktop'],
        { encoding: 'utf8', shell: process.platform === 'win32', env: { ...process.env, REVIEW_RUN: '1' } });
      // Parse flow-level JSON log lines each spec emits; enforce 60s budget
      // Push to flows[]; any elapsedMs > 60_000 or !passed → G15 failure entry
      gates_summary['G15'] = { observed: `${flows.filter(f => f.passed && f.elapsedMs <= 60000).length}/7`, threshold: '7/7 ≤ 60s', passed: flows.length === 7 && flows.every(f => f.passed && f.elapsedMs <= 60_000) };

      // Stage 4 — Lighthouse CI (G9..G12)
      const lhciUrls = ROUTES.slice(0, 10).map((r) => `http://localhost:3000${r.path === '/' ? '' : r.path}`);
      const lhci = spawnSync('npx', ['lhci', 'autorun', ...lhciUrls.map((u) => `--collect.url=${u}`)],
        { encoding: 'utf8', shell: process.platform === 'win32' });
      // Parse .review/lhci/*.json for per-URL LCP/TTI/CLS/JS-size; populate gates G9..G12

      const report: AggregateReport = {
        iteration: iter,
        started_at: started,
        ended_at: new Date().toISOString(),
        failures,
        flows,
        gates_summary,
      };
      mkdirSync('.review', { recursive: true });
      writeFileSync('.review/aggregate-report.json', JSON.stringify(report, null, 2));
      return report;
    }

    async function main() {
      // Standalone entrypoint used by verification 07.1-02-01 and --dry-run flag
      const dryRun = process.argv.includes('--dry-run');
      if (dryRun) {
        const scenarios = ROUTES.length * 2;  // × 2 viewports
        console.log(`[full-sweep:dry-run] scenarios=${scenarios} flow_slots=7`);
        process.exit(scenarios === 108 ? 0 : 1);

      const report = await runFullSweep(0);
      const flowsOk = report.flows.every((f) => f.passed && f.elapsedMs <= 60_000);
      console.log(`[full-sweep] iter=0 failures=${report.failures.length} flows_ok=${report.flows.filter(f=>f.passed).length}/${report.flows.length}`);
      process.exit(report.failures.length === 0 && flowsOk ? 0 : 1);
    }

    if (require.main === module) main().catch((err) => { console.error(err); process.exit(1); });

    export async function runAutoFixLoop(
      generateFixes: (failures: GateFailure[]) => Promise<FixProposal[]>,
      options: { dryRun?: boolean } = {},
    ): Promise<{ outcome: AutoFixOutcome; iterations: IterationLog[] }> {
      const MAX_ITER_STRICT = 3;                   // D-21 default
      const MAX_ITER_PROGRESS = 10;                // D-21 progress-aware extension
      const iterations: IterationLog[] = [];

      if (options.dryRun) {

        console.log('[auto-fix:dry-run] DENY:',     DENYLIST_PATTERNS.map(String));
        return { outcome: 'green', iterations };

      let prev = report.failures.length;
      let noProgress = 0;
      let iter = 0;

      while (report.failures.length > 0) {
        iter++;
        if (iter > MAX_ITER_PROGRESS) return { outcome: 'exhausted', iterations };

        const start = Date.now();
        const proposals = await generateFixes(report.failures);

        const accepted: FixProposal[] = [];
        const rejected: FixProposal[] = [];
        for (const p of proposals) {
          if (isAllowedPath(p.file)) accepted.push(p);
          else rejected.push(p);
        }

        // D-20 escalation log
        for (const r of rejected) {
          appendFileSync('MANUAL-FIX-NEEDED.md',
            `- [${r.file}] ${r.reason} — blocked by D-20; ${r.gate} gate${r.route ? ` at ${r.route}` : ''}\n`);

        for (const p of accepted) writeFileSync(p.file, p.newContent);
        if (accepted.length > 0) {
          execSync(`git add -A && git commit -m "fix(07.1-02): iter ${iter} — ${accepted.length} fixes, ${rejected.length} escalated [iter=${iter}]"`,
            { stdio: 'inherit' });
        }

        report = await runFullSweep(iter);
        const curr = report.failures.length;
        const strict = curr < prev;
        const log: IterationLog = { n: iter, prev, curr, strict, accepted: accepted.length, rejected: rejected.length, duration_ms: Date.now() - start };
        iterations.push(log);
        console.log(`iter=${iter} failures={${prev}→${curr}} progress=${strict ? 'yes' : 'no'} accepted=${accepted.length} rejected=${rejected.length}`);
        prev = curr;

        if (!strict) noProgress++;
        else noProgress = 0;

        if (iter >= MAX_ITER_STRICT && !strict) return { outcome: 'paused', iterations };   // D-21 default pause
        if (noProgress >= 3)                    return { outcome: 'paused', iterations };   // D-21 3-no-progress pause
      }
      return { outcome: 'green', iterations };

    // looping even when G1..G12 report zero. If a D-11c content assertion OR D-13 CTA probe failed,
    // report still FAILs.
    export function hasFunctionalFailure(report: AggregateReport): boolean {

      const result = await runAutoFixLoop(
        async (_failures) => {
          // Default generator: no-op placeholder. A real implementation is Claude's Discretion (D-22).
          // For the dry-run verification this is never invoked.
          return [];
        },

      );

      process.exit(result.outcome === 'green' ? 0 : result.outcome === 'paused' ? 2 : 1);
    }

    if (require.main === module) main().catch((err) => { console.error(err); process.exit(1); });
