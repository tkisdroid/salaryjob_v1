"use server";

import { z } from "zod";
import { revalidatePath } from "next/cache";
import { requireBusiness } from "@/lib/dal";
import { prisma } from "@/lib/db";
import type { ProfileFormState } from "@/lib/form-state";

const JOB_CATEGORIES = [
  "food", "retail", "logistics", "office",
  "event", "cleaning", "education", "tech",
] as const;

const BizProfileSchema = z.object({
  profileId: z.string().uuid("사업장 ID가 올바르지 않습니다"),
  name: z.string().min(1, "상호명은 필수입니다").max(100, "상호명은 100자 이하여야 합니다"),
  category: z.enum(JOB_CATEGORIES, { message: "카테고리를 선택해주세요" }),
  logo: z.string().max(10, "로고 이모지는 10자 이하여야 합니다").optional().or(z.literal("")),
  address: z.string().min(1, "주소는 필수입니다").max(200, "주소는 200자 이하여야 합니다"),
  addressDetail: z.string().max(100, "상세주소는 100자 이하여야 합니다").optional().or(z.literal("")),
  lat: z.coerce
    .number()
    .min(-90, "위도는 -90 이상이어야 합니다")
    .max(90, "위도는 90 이하여야 합니다"),
  lng: z.coerce
    .number()
    .min(-180, "경도는 -180 이상이어야 합니다")
    .max(180, "경도는 180 이하여야 합니다"),
  description: z.string().max(500, "설명은 500자 이하여야 합니다").optional().or(z.literal("")),
});

/**
 * updateBusinessProfile — Business profile CRUD Server Action.
 *
 * BIZ-01: persists name, address, category, logo emoji, description, lat/lng.
 * BIZ-02: read-only fields (rating, reviewCount, completionRate, verified)
 *         are NOT read from FormData.
 * BIZ-03: owner check — the profileId from FormData must belong to the
 *         authenticated session.id, or the action refuses.
 *
 * Supports 1:many: user can edit any BusinessProfile they own, addressed by profileId.
 */
export async function updateBusinessProfile(
  _prevState: ProfileFormState,
  formData: FormData,
): Promise<ProfileFormState> {
  const session = await requireBusiness();

  const raw = {
    profileId: (formData.get("profileId") ?? "") as string,
    name: (formData.get("name") ?? "") as string,
    category: (formData.get("category") ?? "") as string,
    logo: (formData.get("logo") ?? "") as string,
    address: (formData.get("address") ?? "") as string,
    addressDetail: (formData.get("addressDetail") ?? "") as string,
    lat: (formData.get("lat") ?? "") as string,
    lng: (formData.get("lng") ?? "") as string,
    description: (formData.get("description") ?? "") as string,
  };

  const parsed = BizProfileSchema.safeParse(raw);
  if (!parsed.success) {
    const fieldErrors: Record<string, string> = {};
    for (const issue of parsed.error.issues) {
      const key = issue.path[0]?.toString() ?? "form";
      if (!fieldErrors[key]) fieldErrors[key] = issue.message;
    }
    const firstError = Object.values(fieldErrors)[0] ?? "입력값이 올바르지 않습니다";
    return { error: firstError, fieldErrors };
  }

  const d = parsed.data;

  // Owner check — BIZ-03 enforcement at the application layer (Prisma bypasses RLS)
  const existing = await prisma.businessProfile.findUnique({
    where: { id: d.profileId },
    select: { id: true, userId: true },
  });
  if (!existing) {
    return { error: "사업장을 찾을 수 없습니다" };
  }
  if (existing.userId !== session.id) {
    console.warn(
      `BIZ-03 owner check failed: user ${session.id} tried to update profile ${d.profileId} owned by ${existing.userId}`,
    );
    return { error: "이 사업장을 수정할 권한이 없습니다" };
  }

  try {
    // Step 1: Update scalar columns via Prisma
    await prisma.businessProfile.update({
      where: { id: d.profileId },
      data: {
        name: d.name,
        category: d.category,
        logo: d.logo || null,
        address: d.address,
        addressDetail: d.addressDetail || null,
        lat: d.lat,
        lng: d.lng,
        description: d.description || null,
      },
    });

    // Step 2: Update the PostGIS geography(Point) column via raw SQL
    // (Prisma cannot write Unsupported columns; same pattern as Phase 2 seed.ts)
    await prisma.$executeRaw`
      UPDATE public.business_profiles
      SET location = ST_SetSRID(ST_MakePoint(${d.lng}, ${d.lat}), 4326)::geography
      WHERE id = ${d.profileId}::uuid
    `;

    revalidatePath("/biz/profile");
    revalidatePath("/biz");

    return {
      success: true,
      data: { id: d.profileId },
      message: "사업장 정보가 저장되었습니다",
    };
  } catch (e) {
    console.error("updateBusinessProfile error", e);
    return { error: "저장에 실패했습니다. 잠시 후 다시 시도해주세요" };
  }
}

| Client form → updateBusinessProfile Server Action | Untrusted FormData — only whitelisted scalars + profileId pass the Zod schema |

| Server Action → prisma.$executeRaw (geography) | Raw SQL with parameterized values ($1=lng, $2=lat) — Prisma escapes values, no injection |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | Tampering | profileId spoofing — user submits profileId belonging to someone else | mitigate | Explicit owner check: `prisma.businessProfile.findUnique({where:{id:profileId}})` then compare `existing.userId !== session.id` → return error. Prisma bypasses RLS, so this application-layer check is the primary defense. Static test asserts the check string exists in source. |
| T-03-04-02 | Elevation of Privilege | FormData.rating / reviewCount / completionRate / verified | mitigate | Zod schema omits these fields. safeParse() drops unknown keys (z.object does NOT use .passthrough()). Static test asserts `formData.get("rating")` etc. never appear in actions.ts. |
| T-03-04-03 | Tampering | FormData.userId — attempt to re-own profile | mitigate | Zod schema does not include userId. Prisma update does not set userId. userId is immutable once created — by design. |
| T-03-04-04 | Injection | PostGIS $executeRaw with lat/lng | mitigate | Uses Prisma tagged template `prisma.$executeRaw\`...${d.lng}...\`` — values are parameterized. d.lng is already numeric (Zod coerce + number validation), so even if parameterization failed, the value is a float. |
