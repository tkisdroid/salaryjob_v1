      duties: [],
      requirements: [],
      whatToBring: [],

/**
 * Clean up all test jobs created by createTestJob (title starts with "TEST_").
 */
export async function cleanupTestJobs() {
  await prisma.job.deleteMany({
    where: { title: { startsWith: "TEST_" } },
  });
}

/**
 * Set a job's location geography column via raw SQL (Prisma can't write Unsupported).
 * Required for PostGIS distance tests.
 */
export async function setJobLocation(jobId: string, lng: number, lat: number) {
  await prisma.$executeRaw`
    UPDATE public.jobs
    SET location = ST_SetSRID(ST_MakePoint(${lng}, ${lat}), 4326)::geography
    WHERE id = ${jobId}::uuid
  `;
}

Mitigated by: (a) Supabase storage upload is a 1-hop FormData pass-through, not an in-memory buffer; (b) only authenticated Server Actions accept FormData (DAL enforces auth before body parse); (c) file size is also validated in Server Action after parse.
